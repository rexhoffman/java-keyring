145   private void init() throws IOException {
146     try {
147       encryption = new TransportEncryption();
148       encryption.initialize();
149       encryption.openSession();
150       encryption.generateSessionKey();
151       service = encryption.getService();
152       session = service.getSession();
153       prompt = new Prompt(service);
154       withoutPrompt = new InternalUnsupportedGuiltRiddenInterface(service);
155     } catch (DBusException | NoSuchAlgorithmException | InvalidAlgorithmParameterException | InvalidKeySpecException
156         | InvalidKeyException ex) {
157       // log.error(e.toString(), e.getCause());
158       throw new IOException(ex.toString(), ex.getCause());
159     }
160   }

203   private void performPrompt(ObjectPath path) {
204     if (!("/".equals(path.getPath()))) {
205       prompt.await(path);
206     }
207   }
208 
209   private void unlock() {
210     if (collection.isLocked()) {
211       if (encrypted == null) {
212         Pair<List<ObjectPath>, ObjectPath> response = service.unlock(Arrays.asList(collection.getPath()));
213         performPrompt(response.b);
214       } else {
215         withoutPrompt.unlockWithMasterPassword(collection.getPath(), encrypted);
216       }
217     }
218   }

278   /**
279    * Creates an item with the provided properties in this collection.
280    *
281    * @param label      The displayable label of the new item
282    * @param password   The password of the new item
283    * @param attributes The attributes of the new item
284    *
285    * @return DBus object path
286    *
287    * @throws IllegalArgumentException if inputs are null
288    */
289   @SuppressWarnings("rawtypes")
290   public String createItem(String label, CharSequence password, Map<String, String> attributes)
291       throws IllegalArgumentException {
292 
293     if (password == null) {
294       throw new IllegalArgumentException("The password may not be null.");
295     }
296     if (label == null) {
297       throw new IllegalArgumentException("The label of the password may not be null.");
298     }
299 
300     unlock();
301 
302     DBusPath item = null;
303     final Map<String, Variant> properties = Item.createProperties(label, attributes);
304     try (final Secret secret = encryption.encrypt(password)) {
305       Pair<ObjectPath, ObjectPath> response = collection.createItem(properties, secret, false);
306       item = response.a;
307       if ("/".equals(item.getPath())) {
308         Completed completed = prompt.await(response.b);
309         if (!completed.dismissed) {
310           Collection.ItemCreated ic = collection.getSignalHandler().getLastHandledSignal(Collection.ItemCreated.class);
311           item = ic.item;
312         }
313       }
314     } catch (NoSuchAlgorithmException | NoSuchPaddingException | InvalidAlgorithmParameterException
315         | InvalidKeyException | BadPaddingException | IllegalBlockSizeException ex) {
316         // log.error(e.toString(), e.getCause());
317     }
318 
319     return item.getPath();
320   }

336   /**
337    * Updates an item with the provided properties.
338    *
339    * @param objectPath The DBus object path of the item
340    * @param label      The displayable label of the new item
341    * @param password   The password of the new item
342    * @param attributes The attributes of the new item
343    *
344    * @throws IllegalArgumentException if inputs are null
345    */
346   public void updateItem(String objectPath, String label, CharSequence password, Map<String, String> attributes)
347       throws IllegalArgumentException {
348 
349     if (objectPath == null) {
350       throw new IllegalArgumentException("The object path of the item may not be null.");
351     }
352 
353     unlock();
354 
355     Item item = getItem(objectPath);
356 
357     if (label != null) {
358       item.setLabel(label);
359     }
360 
361     if (attributes != null) {
362       item.setAttributes(attributes);
363     }
364 
365     if (password != null) {
366       try (Secret secret = encryption.encrypt(password)) {
367         item.setSecret(secret);
368       } catch (NoSuchAlgorithmException | NoSuchPaddingException | InvalidAlgorithmParameterException
369           | InvalidKeyException | BadPaddingException | IllegalBlockSizeException ex) {
370           // log.error(e.toString(), e.getCause());
371       }
372     }
373   }

414   public List<String> getItems(Map<String, String> attributes) {
415     unlock();
416 
417     List<ObjectPath> objects = collection.searchItems(attributes);
418 
419     if (objects != null && !objects.isEmpty()) {
420       return Static.Convert.toStrings(objects);
421     } else {
422       return null;
423     }
424   }

426   /**
427    * Get the secret of the item.
428    *
429    * @param objectPath The DBus object path of the item
430    *
431    * @return plain chars
432    */
433   public char[] getSecret(String objectPath) {
434     unlock();
435 
436     final Item item = getItem(objectPath);
437 
438     char[] decrypted = null;
439     try (final Secret secret = item.getSecret(session.getPath())) {
440       decrypted = encryption.decrypt(secret);
441     } catch (NoSuchPaddingException | NoSuchAlgorithmException | InvalidAlgorithmParameterException
442         | InvalidKeyException | BadPaddingException | IllegalBlockSizeException ex) {
443         // log.error(e.toString(), e.getCause());
444     }
445     return decrypted;
446   }
